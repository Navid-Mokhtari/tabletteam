 80 /**
 81  * @name call
 82  * @namespace
 83  * @property {Number} id Same as {@link call#callId}.
 84  * @property {Number} callId Unique identifier of the call.
 85  * @property {Boolean} exists True if the call exists.
 86  * @property {Boolean} audioMuted Audio is muted.
 87  * @property {Boolean} videoMuted Video is muted.
 88  * @property {String} state Same as {@link call#callState}.
 89  * @property {String} callState Current state of the call.  May be one of the following:<ul>
 90  * <li>"OffHook" - receiver lifted in CTI control mode, or preparing to make a call in either mode.</li>
 91  * <li>"OnHook" - call is ended or about to go OffHook. If !capabilities.canOriginateCall, then call is ended.</li>
 92  * <li>"Ringout" - remote party is ringing.</li>
 93  * <li>"Ringin" - incoming call.</li>
 94  * <li>"Proceed" - if already on a call in CTI mode and there's an incoming call it may be in "Proceed" state.</li>
 95  * <li>"Connected" - call is connected.</li>
 96  * <li>"Hold" - call is held.</li>
 97  * <li>"RemHold" - call is held on a shared-line device.</li>
 98  * <li>"Resume" - n/a.</li>
 99  * <li>"Busy" - remote line busy.</li>
100  * <li>"Reorder" - call failed.</li>
101  * <li>"Conference"</li>
102  * <li>"Dialing" - dialing number.</li>
103  * <li>"RemInUse" - call on a shared-line device.</li>
104  * <li>"HoldRevert"</li>
105  * <li>"Whisper"</li>
106  * <li>"Parked"</li>
107  * <li>"ParkRevert"</li>
108  * <li>"ParkRetrieved"</li>
109  * <li>"Preservation" - call is in preservation mode.</li>
110  * <li>"WaitingForDigits"</li
111  * ><li>"Spoof_Ringout"</li></ul>
112  * @property {Number} numericState Numeric representation of callState - subject to change, do not use.
113  * @property {String} callType One of the following: <ul><li>"None"</li><li>"Incoming"</li><li>"Outgoing"</li><li>"Forwarded"</li></ul>
114  * @property {String} calledPartyDirectoryNumber <b>Deprecated</b> (since 3.0.2) Use participants and localParticipant instead. <br>
115  * The translated directory number of the called party.
116  * @property {String} calledPartyNumber <b>Deprecated</b> (since 3.0.2) Use participants and localParticipant instead. <br>
117  * The number of the called party.
118  * @property {String} calledPartyName <b>Deprecated</b> (since 3.0.2) Use participants and localParticipant instead. <br>
119  * The name of the called party. 
120  * @property {String} callingPartyDirectoryNumber <b>Deprecated</b> (since 3.0.2) Use participants and localParticipant instead. <br>
121  * The translated directory number of the calling party. 
122  * @property {String} callingPartyNumber <b>Deprecated</b> (since 3.0.2) Use participants and localParticipant instead. <br>
123  * The number of the calling party.
124  * @property {String} callingPartyName <b>Deprecated</b> (since 3.0.2) Use participants and localParticipant instead. <br>
125  * The name of the calling party.
126  * @property {Number} lineId Not yet implemented.
127  * @property {String} mediaState Not implemented.
128  * @property {String} videoDirection One of the following: <ul><li>"Inactive"</li><li>"SendRecv"</li><li>"SendOnly"</li><li>"RecvOnly"</li></ul>
129  * @property {Boolean} isConference true if the call is a conference call.
130  * @property {Number} maxParticipants Maximum number of participants on a conference call.  This parameter is not set if the call is not a conference.
131  * @property {callcapability[]} capabilities A map that has capability name as the key and true as the value. 
132  * Call capabilities indicate operations you can perform on the call, and they should be used to control enabling/disabling/state/access to parts of the UI. Important capabilities are:
133  * <ul>
134  * <li>canOriginateCall - can create new call</li>
135  * <li>canAnswerCall - call can be answered</li>
136  * <li>canHold - call can be Held</li>
137  * <li>canResume - call can be Resumed</li>
138  * <li>canEndCall - call can be Ended</li>
139  * <li>canSendDigit - can send a DTMF digit to the call</li>
140  * <li>canDirectTransfer - can tranfer call</li>
141  * <li>canJoinAcrossLine - can join this call with another call on the same line with this capability also set</li>
142  * <li>canImmediateDivert - can iDivert (e.g. to voicemail)</li>
143  * <li>canUpdateVideoCapability - can update the videoDirection on the call. See {@link $.fn.cwic-updateConversation} examples.</li>
144  * <li>canMuteAudio - can mute audio on the call</li>
145  * <li>canUnmuteAudio - can unmute audio on the call</li>
146  * <li>canMuteVideo - can mute video on the call</li>
147  * <li>canUnmuteVideo - can unmute video on the call</li>
148  * </ul>
149  * @property {participant[]} participants Array of all remote participants on the call.
150  * @property {participant} participant Remote participant (first entry in participants list).
151  * @property {participant} localParticipant Local participant (your line info).
152  * @property {String} callEvent Since: 2.1.1 <br>
153  * The last call event to occur. The callEvent property can be used to check for error conditions, or to determine if call attributes have changed. 
154  * Important events are:<ul>
155  * <li>"evTransferOrConferenceCancelled" - Indicates that attempt to join a conference, or to transfer a call, has been cancelled or has failed.</li>
156  * <li>"evPreservation" - Indicates that a call is preserved because the local endpoint has lost call control, e.g. lost connection to CUCM.</li>
157  * <li>"evConferenceParticipantInfoChanged" - For conference calls you will receive this event if the contents of the participant list changes,
158  * e.g. a new participant added or removed from the list of participants. Note that this occurs only for conference calls in the Connected state.</li>
159  * </ul>
160  */
